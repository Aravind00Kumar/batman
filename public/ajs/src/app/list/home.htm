List Component Example

Installation and usage

  • Add the following scripts and styles in index.html file
    <link rel="stylesheet" href="./node_modules/@batman/core-es5/batman.css">
    <script src="./node_modules/@batman/core-es5/polyfills.js">
    <script src="./node_modules/@batman/core-es5/core.js">
    <script src="./node_modules/@batman/components-es5/list.js">
    <script src="./node_modules/angular/angular.js">
    <script src="./node_modules/@batman/ajs/module.js">
    <script src="./node_modules/@batman/ajs/list.js">
  • In html add the following code
    <bat-list options="listBasicCtrl.listOptions">
  • In controller add the following jscode
    angular.module('app').controller("app.listBasicController", [function () {
    var dataSource = [{ "text": "0" }, { "text": "1" }, { "text": "2" }];
    this.listOptions = {
    data: dataSource,
    height: 30,
    pageSize: 5
    });
    }]);
